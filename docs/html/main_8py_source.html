32def select_model(file_path: str, prompt: str) -> str:
33 """
34 Selects the appropriate model based on routing rules.
35
36 @param file_path Path of the file for which a model is selected
37 @param prompt User's input prompt
38 @return Name of the selected model
39 """
40 ext = file_path.split('.')[-1].lower()
41 prompt = prompt.lower()
42
43 for rule in ROUTING:
44 exts = rule["match"].get("file_extension", [])
45 kws = rule["match"].get("prompt_contains", [])
46 if (not exts or ext in exts) and (not kws or any(k in prompt for k in kws)):
47 return rule["route_to"]
48
49 return "qwen3:4b"
